Tree Shaking 按需导入（推荐）

import "yn-web-component/components/yn-pull-cord-switch";
import { applyPullCordShellBackground } from "yn-web-component/components/yn-pull-cord-switch";

const shell = document.querySelector("#lamp-shell");
const lamp = document.querySelector("yn-pull-cord-switch");

applyPullCordShellBackground(shell, lamp);

lamp.addEventListener("change", (e) => {
  applyPullCordShellBackground(shell, lamp, e.detail.checked);
});
